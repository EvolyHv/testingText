def onehot_to_text(mat,tokenizer, cutoff):
    mat = pd.DataFrame(mat)
    mat.rename(columns=tokenizer.index_word, inplace=True)
    output = mat.sum(axis=1)
    for row in range(mat.shape[0]):
        if output[row] == 0:
            output[row] = []
        else:
            output[row] = mat.columns[mat.iloc[row,:] >= cutoff].tolist()
    return(output)

def getSetFromIndexes(wordIndexes, xLen, step): #    ,  ,  
    xSample = [] #    
    wordsLen = len(wordIndexes) #   
    index = 0 #    

    while (index + xLen <= wordsLen):#      
        xSample.append(wordIndexes[index:index+xLen]) # ""   xLen
        index += step #    step
    
    return xSample

def createSetsMultiClasses(wordIndexes, xLen, step): #    ,  ,  

    #    6 
    #  /   
    nClasses = len(wordIndexes) #    
    classesXSamples = []        #     "- *-   *  (, 6  1341*1000)"
    for wI in wordIndexes:      #       
        classesXSamples.append(getSetFromIndexes(wI, xLen, step)) #      ,   "- * " 

    #    xSamples
    xSamples = [] #     " -    *  (, 15779*1000)"
    ySamples = [] #     " -    *  6"

    for t in range(nClasses): #   - (6)
        xT = classesXSamples[t] #     "-   * "(, 1341*1000)
        for i in range(len(xT)): #    
            xSamples.append(xT[i]) #     
            ySamples.append(utils.to_categorical(t, nClasses)) #    

    xSamples = np.array(xSamples) #    numpy    
    ySamples = np.array(ySamples) #    numpy    

  
    return (xSamples, ySamples) #
